§4 Ws* 调优 + 新原语 API 规格

这一段以"代码契约"为主。每个原语给 props / events / slots / 用例;现有 Ws* 只列"新增/调优"部分,不动部分省略。

§4.1 现有 Ws* 调优(仅列变更项)

▌ WsButton · 新增 2 个 variant + 1 个 prop

// 新增 variant
variant: 'primary' | 'secondary' | 'ghost' | 'danger'
       | 'primary-gradient'   // 渐变紫→青,用于"告诉 AI Agent"、"▶ 运行"等主 CTA
       | 'accent'            // 单色 accent,用于"▶ Dry Run"等次要操作
// 新增 prop
iconOnly?: boolean           // 正方形按钮,仅图标(28×28 / 32×32)
loading?: boolean            // 显示 spinner,自动 disable
// 微调
height: 默认 32px (was 36px)        // 适配密集工具栏
height-sm: 26px (was 28px)          // 用于 popover 内 / 行内按钮
focus-ring: 用 --shadow-focus-ring(原已支持,统一调用)

▌ WsCard · 新增 2 个 variant + 微调 hover

// 新增 variant
variant: 'default' | 'elevated' | 'outlined'
  - default:  现有样式(浅阴影 + border),保持
  - elevated: 更深阴影 + hover translateY(-2px) + 紫色光晕,用于工作流卡
  - outlined: 无阴影,仅 border,用于编辑器三栏面板
// 微调
hoverable?: boolean          // true 时启用 --shadow-card-hover + transform
padding: 'sm' | 'md' | 'lg' | 'none'   // 取代硬编码 padding,默认 md

▌ WsEmpty · 新增 2 个 preset

// preset 新增(原有: 'no-data' | 'no-results' | 'error' | 'no-permission')
preset: 'no-data' | 'no-results' | 'error' | 'no-permission'
      | 'no-workflow'           // 空工作流列表:装饰图 + 双 CTA(主:告诉 AI Agent / 次:从空白开始)
      | 'workflow-search-empty' // 搜索无结果:🔍 + "清空搜索"按钮
// 新增 slot
#illustration                 // 自定义装饰图(默认用 preset 自带 SVG)
#extra                        // preset 下方的额外内容(按钮组 / 提示等)

▌ WsPopover · 确保 dropdown 场景

// 新增 / 强化
placement: 'bottom-start' | 'bottom-end' | ...(原 12 种扩展为 16)
trigger: 'click' | 'hover' | 'manual'   // 已有
offset: number = 6                       // 已有,默认值从 8 调为 6
// dropdown 场景推荐
<WsPopover placement="bottom-end" :offset="6" trigger="click">
  <template #trigger> <WsButton variant="primary" icon-only>+</WsButton> </template>
  <WsMenu :items="..." />
</WsPopover>
// 已有遮罩/ESC 关闭,无需新增

§4.2 新增 4 个原语 API 规格

▌ WsNodeChip · 节点类型紧凑标签

// 用途:NodePalette 每条目 / 节点元信息展示
interface WsNodeChipProps {
  type: NodeType                        // 14 种内置类型之一(start/end/pivot/condition/skip/...)
  label?: string                        // 默认 = type;支持覆盖
  description?: string                  // 1 行简短描述(显示在 label 下方)
  size?: 'sm' | 'md'                    // 默认 sm(用于 palette),md(用于卡片)
  iconOnly?: boolean                    // 仅图标
  draggable?: boolean                   // 启用原生拖拽;触发 dragstart 事件
}
// Events
(e: 'dragstart', native: DragEvent): void
// Slots
#default                                // 覆盖 label 文案
#icon                                   // 覆盖默认图标
#meta                                   // label 右侧的额外元(如计数/快捷键)
// 内部实现
- icon: 从 node-metadata.ts.builtinNodeMetadata[type].icon 读
- color: 从 type 映射 6 种分类色(执行=蓝/决策=橙/控制=灰/触发=青/AI=紫/数据回流=琥珀)
- 容器: 22×22 (sm) / 32×32 (md) icon-only 模式;有 label 时 36px 高 (sm) / 44px 高 (md)

▌ WsStatusDot · 运行状态点

// 用途:工作流卡状态 / 画布节点运行态 / 时间线 / 列表行
type RunStatus = 'idle' | 'running' | 'paused' | 'completed'
              | 'failed' | 'skipped' | 'awaiting-decision'
interface WsStatusDotProps {
  status: RunStatus
  size?: 'xs' | 'sm' | 'md'             // 6px / 8px / 12px
  pulse?: boolean                       // true 时显示呼吸光环(running / awaiting-decision 默认 true)
  showLabel?: boolean                   // true 时右侧带文字(用 i18n 字典)
  label?: string                        // 覆盖默认文字
}
// 默认颜色映射
- idle               → --color-text-tertiary  (灰)
- running            → --color-warning        (琥珀,带 pulse)
- paused             → --color-info           (蓝)
- completed          → --color-success        (绿)
- failed             → --color-danger         (红)
- skipped            → --color-text-tertiary  (灰)
- awaiting-decision  → --color-primary        (紫,带 pulse,最显眼)

▌ WsMiniBadge · 小徽章

// 用途:工作流卡分类标签 / 节点调色板项的次要信息 / 表单字段计数
interface WsMiniBadgeProps {
  variant?: 'neutral' | 'primary' | 'accent' | 'success' | 'warning' | 'danger'
  size?: 'sm' | 'md'                    // sm = 10px font, md = 11px font
  pill?: boolean                        // 圆角 99px(默认 true)
  icon?: string                         // 前置 emoji / lucide name
}
// Slot
#default                                // 文本(支持 :count 形式)
// 视觉
- 背景: variant × 12% 透明
- 文字: variant × 100%
- 边框: variant × 30% 透明, 1px
- 高度: sm=20px, md=24px
- padding: sm=3px 8px, md=4px 10px

▌ WsSegmentedControl · 分段切换

// 用途:EditorModeToggle (简化/高级) / 视图模式切换
interface SegmentedOption {
  value: string
  label: string
  icon?: string                         // lucide name
  disabled?: boolean
}
interface WsSegmentedControlProps {
  modelValue: string
  options: SegmentedOption[]            // 2-5 段
  size?: 'sm' | 'md'                    // sm=24px 高, md=28px 高
  block?: boolean                       // 撑满父容器(等分)
}
// Events
(e: 'update:modelValue', value: string): void
// 视觉
- 容器: 1px border, --color-border, 6px radius
- 选中段: 背景 --color-primary-subtle, 文字 --color-primary, font-weight 600
- 悬浮未选中: 背景 rgba(255,255,255,.04)
- 过渡: 100ms ease
